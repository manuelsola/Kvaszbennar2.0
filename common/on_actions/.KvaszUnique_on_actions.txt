#When adding effects directly in on_actions always consider if it would be best to do so through a scripted effect. on_actions with a lot of script in them are harder to get an overview of.

# country
on_startup = {
	#### Dev click counter ###
	#ARUMBA LINES
	every_country = {
		if = {
			limit = {
				check_variable = {
					which = dev_adm_clicks					
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dev_adm_clicks
				value = 0
			}
		}
	}
	
	every_country = {
		if = {
			limit = {
				check_variable = {
					which = dev_dip_clicks					
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dev_dip_clicks
				value = 0
			}
		}
	}
	
	every_country = {
		if = {
			limit = {
				check_variable = {
					which = dev_mil_clicks					
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dev_mil_clicks
				value = 0
			}
		}
	}
	
	every_country = {
		if = {
			limit = {
				check_variable = {
					which = dev_sum_clicks					
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dev_sum_clicks
				value = 0
			}
		}
	}
}


# province
on_adm_development = {
	#ARUMBA LINES dev clicks counter
	hidden_effect = {
		owner = {
			change_variable = {
				which = dev_adm_clicks
				value = 1
			}
			change_variable = {
				which = dev_sum_clicks
				value = 1
			}
		}
	}
}
